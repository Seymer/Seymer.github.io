String.prototype.endsWith   =
function ( str )
{
    if ( str == null || str.length > this.length )
    {
        return false;
    }
    if ( this.substring( this.length - str.length ) == str )
    {
        return true;
    }
    else
    {
        return false;
    }
};

#### 用indexOf()实现
#+BEGIN_EXAMPLE
String.prototype.endsWith   =
function ( str )
{
    return this.indexOf( str, this.length - str.length ) !== -1;
};
#+END_EXAMPLE 

#### 用substr()实现
#+BEGIN_EXAMPLE
String.prototype.endsWith   =
function ( str )
{
    return this.substr( -str.length ) === str;
};
#+END_EXAMPLE
